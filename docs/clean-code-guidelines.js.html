        this.email = userData.email;
    }

        this.id = 'ui';
    }
}

/**
 * Sum up numbers
 * @param  {...number} numbers 
 * @returns sum of all the numbers given
 */
function sumUp(...numbers){
    let sum = 0;
    for(const number of numbers){
        sum += number;
    }
    return sum;
}

const total = sumUp(10,24,-31);
total;
